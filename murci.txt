    <script>
        // Smooth scrolling for nav links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', e => {
                e.preventDefault();
                document.querySelector(anchor.getAttribute('href')).scrollIntoView({ behavior: 'smooth' });
            });
        });

        // Animate adventure cards on scroll
        const observer = new IntersectionObserver(entries => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.style.opacity = '1';
                    entry.target.style.transform = 'translateY(0)';
                }
            });
        });
        document.querySelectorAll('.adventure-card').forEach(card => {
            card.style.opacity = '0';
            card.style.transform = 'translateY(20px)';
            card.style.transition = 'opacity 0.5s ease, transform 0.5s ease';
            observer.observe(card);
        });

        // Upload media and add to gallery with delete button
        const mediaUpload = document.getElementById('mediaUpload');
        const uploadBtn = document.getElementById('uploadBtn');
        const galeriaContainer = document.getElementById('galeriaContainer');

        function createMediaElement(file) {
            const url = URL.createObjectURL(file);
            const container = document.createElement('div');
            container.className = 'media-item';

            const deleteBtn = document.createElement('button');
            deleteBtn.className = 'delete-btn';
            deleteBtn.title = 'Eliminar';
            deleteBtn.textContent = '×';
            deleteBtn.addEventListener('click', () => {
                URL.revokeObjectURL(url);
                container.remove();
            });
            container.appendChild(deleteBtn);

            if (file.type.startsWith('image/')) {
                const img = document.createElement('img');
                img.src = url;
                img.alt = 'Foto subida por el usuario - Aventura en Murcia';
                container.appendChild(img);
            } else if (file.type.startsWith('video/')) {
                const video = document.createElement('video');
                video.src = url;
                video.controls = true;
                video.title = 'Video subido por el usuario - Aventura en Murcia';
                container.appendChild(video);
            } else if (file.type.startsWith('audio/')) {
                const audio = document.createElement('audio');
                audio.src = url;
                audio.controls = true;
                audio.title = 'Audio subido por el usuario - Canción de aventura';
                container.appendChild(audio);
            } else {
                alert('Tipo de archivo no soportado: ' + file.type);
                URL.revokeObjectURL(url);
                return null;
            }
            return container;
        }

        uploadBtn.addEventListener('click', () => {
            const files = mediaUpload.files;
            if (!files.length) {
                alert('Por favor, selecciona al menos un archivo.');
                return;
            }
            for (const file of files) {
                const mediaElement = createMediaElement(file);
                if (mediaElement) {
                    galeriaContainer.appendChild(mediaElement);
                }
            }
            mediaUpload.value = '';
        });

        // Add delete functionality to existing images
        document.querySelectorAll('.media-item .delete-btn').forEach(btn => {
            btn.addEventListener('click', e => {
                const container = e.target.parentElement;
                container.remove();
            });
        });
    </script>
